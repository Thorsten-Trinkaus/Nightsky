//set precision
precision mediump float;

//input
attribute vec3 vertPosition;
attribute vec3 vertColor;
//output
varying vec3 fragColor;
//input for both vertex shader and fragment shader
uniform mat4 mWorld;
uniform mat4 mView;
uniform mat4 mProj;

//main
void main()
{
  fragColor = vertColor;
  gl_Position = mProj * mView * mWorld * vec4(vertPosition, 1.0);
}
